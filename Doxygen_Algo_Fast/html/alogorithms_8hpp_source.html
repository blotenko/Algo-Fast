204class mergeSort :public algoSort

206public:
207 void algo(int data[],int lenD)
208 {
209 if(lenD>1){
210 int middle = lenD/2;
211 int rem = lenD-middle;
212 int* L = new int[middle];
213 int* R = new int[rem];
214 for(int i=0;i<lenD;i++){
215 if(i<middle){
216 L[i] = data[i];
